```go
func mockLookPathOK(path string) func(string) (string, error) {
    return func(file string) (string, error) { return path, nil }
}

    return func(file string) (string, error) {
        return "", exec.ErrNotFound
    }
}

// mockCommand returns a command func that returns the given output or error
func mockCommandOK(output string) func(string, ...string) ([]byte, error) {
    return func(name string, args ...string) ([]byte, error) {
        return []byte(output), nil
    }
}

func mockCommandExitError(code int, stderr string) func(string, ...string) ([]byte, error) {
    return func(name string, args ...string) ([]byte, error) {
        // Create an exec.ExitError-like error
        // Since we can't easily construct exec.ExitError, use a helper approach:
        // Run a command that exits with the desired code to get a real ExitError
        cmd := exec.Command("sh", "-c", fmt.Sprintf("echo -n '%s' >&2; exit %d", stderr, code))
        out, err := cmd.Output()
        return out, err
    }
}
```

IMPORTANT: For the ExitError mock, the cleanest approach is to run a real subprocess `sh -c "exit N"` that produces the right ExitError type. This is the standard Go testing pattern for exec.ExitError since you cannot construct one directly.
