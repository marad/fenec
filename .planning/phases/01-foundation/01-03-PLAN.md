    3. Create `internal/repl/repl.go` with the main REPL loop:
       ```go
       package repl

       import (
           "context"
           "fmt"
           "io"
           "os"
           "strings"

           "github.com/chzyer/readline"

           "github.com/marad/fenec/internal/chat"
           "github.com/marad/fenec/internal/config"
           "github.com/marad/fenec/internal/render"
       )

       // REPL manages the interactive chat loop.
       type REPL struct {
           client   chat.ChatService
           conv     *chat.Conversation
           rl       *readline.Instance
           cancelFn context.CancelFunc // For cancelling active generation
       }

       // NewREPL creates a REPL connected to the given chat service.
       func NewREPL(client chat.ChatService, model string, systemPrompt string) (*REPL, error)

       // Run starts the interactive REPL loop. Blocks until exit.
       func (r *REPL) Run() error

       // Close cleans up REPL resources.
       func (r *REPL) Close()
       ```

       **NewREPL implementation:**
       - Get history file path from `config.HistoryFile()`.
       - Create readline instance with `readline.NewEx(&readline.Config{...})`:
         - `Prompt`: `render.FormatPrompt(model)` (per D-01)
         - `HistoryFile`: from config
         - `InterruptPrompt`: `"^C"` (per D-04: Ctrl+C clears input)
         - `EOFPrompt`: `"exit"` (per D-04: Ctrl+D exits)
       - Create `chat.NewConversation(model, systemPrompt)`.

       **Run implementation -- the main REPL loop:**
       ```
       Print startup banner (per D-13): render.FormatBanner(config.Version)
       Print blank line (per D-07)

       Loop:
         1. Read line with rl.Readline()
            - If err == readline.ErrInterrupt: continue (per D-04: Ctrl+C clears input)
            - If err == io.EOF: break (per D-04: Ctrl+D exits)

         2. Handle multi-line input (per D-02):
            - If trimmed line ends with "\", strip backslash, set continuation prompt "... ",
              accumulate lines until one doesn't end with "\"
            - Restore original prompt after multi-line completes

         3. Check if input is a slash command (per D-03):
            - /quit: return nil (exit)
            - /help: print helpText
            - /model: call handleModelCommand (see below)
            - Unknown /command: print "Unknown command: {cmd}. Type /help for available commands."
            - If command handled, continue loop

         4. If input is empty or whitespace, continue

         5. Send message to model:
            a. conv.AddUser(input)
            b. Create cancellable context: ctx, cancel := context.WithCancel(context.Background())
               Store cancel in r.cancelFn (for Escape key cancellation)
            c. Start spinner: sp := render.NewSpinner(rl.Stdout()); sp.Start()
            d. Track raw output for line counting:
               var rawLines int
               var rawContent strings.Builder
            e. Call client.StreamChat(ctx, conv, onToken) where onToken:
               - On first token (use chat.FirstTokenNotifier): sp.Stop()
               - fmt.Fprint(rl.Stdout(), token) -- write to readline's stdout (per Pitfall 2)
               - rawContent.WriteString(token)
            f. After streaming completes (or errors):
               - sp.Stop() (idempotent, in case no tokens arrived)
               - If error is context.Canceled: print "\n[generation cancelled]"
               - If other error: print render.FormatError(err.Error())
               - If success:
                 * Count raw lines: rawLines = render.CountLines(rawContent.String())
                 * Render markdown: rendered = render.RenderMarkdown(rawContent.String(), TerminalWidth())
                 * Overwrite raw output: render.OverwriteRawOutput(rl.Stdout(), rawLines, rendered)
                 * Auto-page if rendered output exceeds terminal height (per D-08):
                   Check if rendered line count > TerminalHeight(). If so, use PageOutput.
                 * conv.AddAssistant(rawContent.String())
            g. Print blank line separator (per D-07)
            h. r.cancelFn = nil
       ```

       **handleModelCommand implementation (per D-09, D-10):**
       ```
       1. Call client.ListModels(ctx)
       2. If error, print error and return
       3. Print numbered list:
          "Available models:"
          "  1. gemma4:latest" (with * prefix if currently active)
          "  2. llama3:latest"
       4. Read selection from rl.Readline() with prompt "Select model [1-N]: "
       5. Parse number, validate range
       6. Switch model: conv.SetModel(selectedModel) (per D-11: history preserved)
       7. Update readline prompt: rl.SetPrompt(render.FormatPrompt(selectedModel))
       8. Print "Switched to {model}"
       ```

    4. Update `main.go` to wire everything together:
       ```go
       package main

       import (
           "context"
           "flag"
           "fmt"
           "os"
           "time"

           "github.com/marad/fenec/internal/chat"
           "github.com/marad/fenec/internal/config"
           "github.com/marad/fenec/internal/render"
           "github.com/marad/fenec/internal/repl"
       )

       func main() {
           // Parse flags (per D-16: --host flag to override default)
           host := flag.String("host", "", "Ollama server address (default: localhost:11434)")
           flag.Parse()

           // Determine host
           ollamaHost := config.DefaultHost
           if *host != "" {
               ollamaHost = *host
           }

           // Create Ollama client
           client, err := chat.NewClient(ollamaHost)
           if err != nil {
               fmt.Fprintln(os.Stderr, render.FormatError(
                   fmt.Sprintf("Failed to create client: %v", err)))
               os.Exit(1)
           }

           // Health check (per D-14: if Ollama unreachable, show error with fix instructions and exit)
           ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
           defer cancel()
           if err := client.Ping(ctx); err != nil {
               fmt.Fprintln(os.Stderr, render.FormatError(
                   fmt.Sprintf("Cannot connect to Ollama at %s. Is it running? Start with: ollama serve\n\nDetails: %v", ollamaHost, err)))
               os.Exit(1)
           }

           // Get available models and select first (per D-09)
           models, err := client.ListModels(ctx)
           if err != nil || len(models) == 0 {
               fmt.Fprintln(os.Stderr, render.FormatError(
                   "No models available. Pull one with: ollama pull gemma4"))
               os.Exit(1)
           }
           defaultModel := models[0]

           // Load system prompt (per D-15)
           systemPrompt, err := config.LoadSystemPrompt()
           if err != nil {
               fmt.Fprintln(os.Stderr, render.FormatError(
                   fmt.Sprintf("Failed to load system prompt: %v", err)))
               os.Exit(1)
           }

           // Create and run REPL
           r, err := repl.NewREPL(client, defaultModel, systemPrompt)
           if err != nil {
               fmt.Fprintln(os.Stderr, render.FormatError(
                   fmt.Sprintf("Failed to start REPL: %v", err)))
               os.Exit(1)
           }
           defer r.Close()

           if err := r.Run(); err != nil {
               fmt.Fprintln(os.Stderr, render.FormatError(err.Error()))
               os.Exit(1)
           }
       }
       ```
